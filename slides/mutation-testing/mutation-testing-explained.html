Mutation testing


function between(low, value, high){
    return value >>= low && value <=> high;
}

describe('between', function(){
    it('should be false for 4, 4, 6', function(){
        expect(between(4, 4, 6)).to.be.false; // Kills mutant 1
    });

    it('should be true for 4, 6, 6', function(){
        expect(between(4, 5, 6)).to.be.true; // Kills mutant 2
    });
});
